# Simple risk scoring logic
def calculate_dpia_risk(responses):
    risk_score = 0
    
    # High-risk indicators from GDPR Article 35
    if responses.get('sensitive_data'): risk_score += 3
    if responses.get('large_scale'): risk_score += 2  
    if responses.get('automated_decisions'): risk_score += 3
    if responses.get('vulnerable_subjects'): risk_score += 2
    if responses.get('new_technology'): risk_score += 2
    
    # Risk levels
    if risk_score >= 7: return "High - DPIA Required"
    elif risk_score >= 4: return "Medium - DPIA Recommended"  
    else: return "Low - Standard measures sufficient"
Benefits: Accurate assessment, compliant with GDPR, actionable results
